  • -isTokenSyncAllowed

    This API is to check whether token sync process is allowed

    @discussion On Debug framework, this API will always return YES, if the token sync slot is not set.

    // For debugging purpose, the MPA could test the token sync feature by modifying the token sync slot.
// Note: It only works on DEBUG framework.
NSInteger weekday = 1;      // 1 for Sunday, ... and 7 for Saturday
NSInteger hour = 0;         // 0 to 23
NSInteger minute = 0;       // 0 to 59
NSInteger second = 0;       // 0 to 59
NSInteger milliSecond = 0;  // 0 to 999
NSString *timeslotString = [NSString stringWithFormat:@"%ld,%ld,%ld,%ld,%ld", weekday, hour, minute, second, milliSecond];

NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
[userDefaults setObject:timeslotString forKey:@"CPS.tokenSyncSlot"];
[userDefaults synchronize];

    5.1.0

  • -getTokenSyncSchedule

    This API will let MPA know when is the next scheduled time slot for tokenSync.

    @discussion On Debug framework, this API will always return the current timestamp, if the token sync slot is not set.

    // For debugging purpose, the MPA could test the token sync feature by modifying the token sync slot.
// Note: It only works on DEBUG framework.
NSInteger weekday = 1;      // 1 for Sunday, ... and 7 for Saturday
NSInteger hour = 0;         // 0 to 23
NSInteger minute = 0;       // 0 to 59
NSInteger second = 0;       // 0 to 59
NSInteger milliSecond = 0;  // 0 to 999
NSString *timeslotString = [NSString stringWithFormat:@"%ld,%ld,%ld,%ld,%ld", weekday, hour, minute, second, milliSecond];

NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
[userDefaults setObject:timeslotString forKey:@"CPS.tokenSyncSlot"];
[userDefaults synchronize];
